## YouTube Music API

import requests

url = "https://youtube-music1.p.rapidapi.com/v2/search"

querystring = {"query":"eminem"}

headers = {
	"X-RapidAPI-Key": "8ca68e698emsh077b34dcb5cedc7p176bfdjsn4a11c2cb1bef",
	"X-RapidAPI-Host": "youtube-music1.p.rapidapi.com"
}

print(response.text)
